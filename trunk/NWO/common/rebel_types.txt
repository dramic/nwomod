populist_rebels = {
	icon = 1
	
	area = nation 					# Stays within the nation

	
	# <this is the governement in the country> = <this is what the rebels want>
	government = {
		proletarian_dictatorship = populist_dictatorship
		presidential_dictatorship = populist_dictatorship
		populist_dictatorship = populist_dictatorship
		islamist_dictatorship = populist_dictatorship
		absolute_monarchy = populist_dictatorship
		nationalist_dictatorship = populist_dictatorship
		hms_government = populist_dictatorship
		democracy = populist_dictatorship
	}
	
	
	defection = none					# Will defect to the "best" alternative.
	independence = none				# Does not strive for independence
	defect_delay = 18
	
	ideology = populist
	
	allow_all_cultures = yes
	allow_all_religions = yes
	allow_all_ideologies = yes
	
	resilient = yes					# Disbands on retreat
	reinforcing = yes				# Does not receive reinforcements
	general = yes					# Does not use generals
	smart = yes						# Does not assault forts, tactically withdraw from battle, etc.
	unit_transfer = no				# Units do not remain after enforcing demands.
		
	occupation_mult = 1.0			# The amount of strength this rebel gains from taking provinces


	# triggers a general rising, modifier will be multiplied with the countrys army strength and compared to the factions strenght
	will_rise = 
	{
		factor = 0.50
		
		modifier = {
			war = yes
			factor = 0.8
		}

		modifier = {
			war_exhaustion = 4
			factor = 0.95
		}	


		modifier = {
			war_exhaustion = 6
			factor = 0.95
		}

		modifier = {
			war_exhaustion = 8
			factor = 0.95
		}
		modifier = {
			factor = 0.85
			average_militancy = 4
		}
		modifier = {
			factor = 0.75
			average_militancy = 6
		}
		modifier = {
			factor = 0.5
			average_militancy = 8
		}
		
	}
	
	# The rebel type with the highest modifier for this pop gets picked
	spawn_chance = {
		factor = 2
		modifier = {
			factor = 0.01
			NOT = {
				OR = {
					populist = 10
					conservative = 10
					islamist = 10
				}
			}
		}
		modifier = {
			factor = 0
			OR = {
				 social_movement = yes
				 political_movement = yes
			}
		}
		modifier = {
			factor = 5
			populist = 10
			NOT = { populist = 20 }
		}
		modifier = {
			factor = 10
			populist = 20
			NOT = { populist = 30 }
		}
		modifier = {
			factor = 15
			populist = 30
			NOT = { populist = 40 }
		}
		modifier = {
			factor = 20
			populist = 40
			NOT = { populist = 50 }
		}
		modifier = {
			factor = 25
			populist = 50
		}
		modifier = {
			factor = 5
			pop_majority_ideology = populist
		}
		modifier = {
			factor = 0
			country = { civilized = no }
		}
		modifier = {
			factor = 5
			NOT = { literacy = 0.30 }
		}
		modifier = {
			factor = 0.5
			literacy = 0.50
			NOT = { pop_majority_ideology = populist }
		}
		modifier = {
			factor = 3
			NOT = { consciousness = 1 }
		}
		modifier = {
			factor = 3
			NOT = { consciousness = 2 }
		}
		modifier = {
			factor = 3
			NOT = { consciousness = 3 }
		}
		modifier = {
			factor = 5
			pop_majority_ideology = conservative
			NOT = {
				country = {
					OR = {
						has_country_modifier = global_liberal_agitation
						has_country_modifier = springtime_of_nations
					}	
				}
			}
		}
		modifier = {
			factor = 0
			is_primary_culture = no
			is_accepted_culture = no
		}
		modifier = {
			factor = 0.01
			is_primary_culture = no
			location = {
				any_core = {
					is_primary_culture = THIS
					is_cultural_union = no
					exists = no
				}
			}
		}
		modifier = {
			factor = 0.01
			is_primary_culture = yes
			location = {
				NOT = { is_core = THIS }
				any_core = {
					is_primary_culture = THIS
					is_cultural_union = no
					exists = no
				}
			}
		}
		modifier = {
			factor = 0.8
			is_accepted_culture = yes
			NOT = { is_culture_group = THIS }
		}
		modifier = {
			factor = 0
			country = { government = absolute_monarchy}
		}
		modifier = {
			factor = 0.2
			country = { government = presidential_dictatorship }
		}
		modifier = {
			factor = 1.5
			country = { plurality = 10 }
		}
		modifier = {
			factor = 0.1
			country = {
				tag = SPA
				OR = {
					government = hms_government
					government = absolute_monarchy
					government = democracy
				}
			}
			OR = {
				populist = 10
				pop_majority_ideology = conservative
			}
		}	
	}
 
 	#This is checked for EACH province in the Area of Operations
	movement_evaluation = {
		factor = 1
		modifier = {
			factor = 0.1
			units_in_province = 1
			NOT = { is_capital = yes }
		}
		modifier = {
			factor = 0.001
			controlled_by = REB
			NOT = { is_capital = yes }
		}
		modifier = {
			factor = 1.5
			average_militancy = 4
		}
		modifier = {
			factor = 1.5
			average_militancy = 6
		}
		modifier = {
			factor = 1.5
			average_militancy = 8
		}
		modifier = {
			factor = 5.0
			is_capital = yes
		}
	}
	
 	# Province Scope
 	siege_won_trigger = {
		NOT = { has_province_modifier = war_torn }
 	}
	
	siege_won_effect = {
		add_province_modifier = {
			name = war_torn
			duration = 365
		}
		any_pop = {
			ideology = {
				value = populist
				factor = 0.05
			}
		}
	}

	# Country scope
	demands_enforced_trigger = {
		OR = {
			capital_scope = {
				controlled_by = REB
				province_control_days = 90
			}
			NOT = {
				any_owned_province = {
					is_colonial = no
					controlled_by = THIS
				}
			}
		}				
	}
	
	demands_enforced_effect = {
		ruling_party_ideology = populist
		political_reform = none_voting
		political_reform = ruling_party_opinion
		political_reform = quality_of_service
		political_reform = censored_publishing
		political_reform = gender_equal
		political_reform = parties_not_allowed
		political_reform = nobody_gets_out
		political_reform = hbtq_outlawed
		political_reform = minorities_oppression
		clr_country_flag = conservative_election_win
		clr_country_flag = socialist_election_win
		clr_country_flag = liberal_election_win
		clr_country_flag = communist_election_win
		clr_country_flag = islamist_election_win
		clr_country_flag = populist_election_win
		clr_country_flag = progressive_election_win
		remove_country_modifier = expert_diplomat
		remove_country_modifier = successful_industrialist
		remove_country_modifier = ivory_tower_intellectual

		remove_country_modifier = lord_admiral

		remove_country_modifier = luddite
		remove_country_modifier = great_statesman
		remove_country_modifier = backroom_boy
		remove_country_modifier = great_reformer
		remove_country_modifier = gauche_buffoon
		remove_country_modifier = administrative_genius
		remove_country_modifier = visionary_thinker

		remove_country_modifier = imperious_autocrat
